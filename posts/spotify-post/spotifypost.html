#Use pandas to identify the ten artists with the most tracks
spotify['track_count'] = spotify.groupby('artist_name')['track_name'].transform('nunique')
top_10_artists = (
    spotify[['artist_name', 'track_count']]
    .drop_duplicates()
    .sort_values(by='track_count', ascending=False)
    .head(10)
)

print(top_10_artists)
